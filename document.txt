class AuthBloc extends Cubit<AuthState> {
  AuthBloc() : super(InitialState());

  String? password;
  String? email;

  void setEmail(String? changeEmail) {
    email = changeEmail;
  }

  void setPassword(String? changePassword) {
    password = changePassword;
  }

  void validate() {
    if (email != null && email != '') {
      emit(ValidateState());
    } else {
      emit(FailureState());
    }
    if (password != null && password != '') {
      print("hhh");
    } else {
      print("hhhddddddddddddddd");
    }
  }
}
